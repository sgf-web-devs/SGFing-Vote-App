NatCam.CaptureFrame

public static bool CaptureFrame (Texture2D frame);

Description

Capture the current preview frame.

You must create the texture to match the width and height of NatCam.Preview. The texture format should be RGBA32. 

void SaveScreenshot () {
    // Create the frame texture
    Texture2D frame = new Texture2D(
        NatCam.Preview.width,
        NatCam.Preview.height,
        TextureFormat.RGBA32,
        false
    );
    // Capture the preview frame
    NatCam.CaptureFrame(frame);
    // Encode it to JPEG
    byte[] jpg = frame.EncodeToJPG();
    // Save it
    System.IO.WriteAllBytes("screenshot.jpg", jpg);
    // Free the texture and its memory
    // Note that this is very important to avoid a memory leak
    Texture2D.Destroy(frame);
}
